    <script>
        // ─────────────────────────────────────────────────────
        // YOUR GOOGLE APPS SCRIPT URL — ALREADY CONNECTED ✅
        // ─────────────────────────────────────────────────────
        const SCRIPT_URL = 'https://script.google.com/macros/s/AKfycbyy6JXmgvGHRI6nUOs0beCEE0pUGFqXFYF0IN_MrWfRwHnAWP22gh2q5IL1mwysyXAF/exec';

        // Lead data store
        const lead = {};
        let currentStepNum = 1;
        const TOTAL_STEPS = 6;

        // ─────────────────────────────────────────────────────
        // ADVANCE to next step (qualified answer)
        // ─────────────────────────────────────────────────────
        function advance(field, value, nextStep) {
            lead[field] = value;
            lead.source = document.referrer || 'Direct';
            showStep('step' + nextStep);
            currentStepNum = nextStep;
            updateProgress(nextStep);
        }

        // ─────────────────────────────────────────────────────
        // DISQUALIFY — save reason, show disq screen
        // ─────────────────────────────────────────────────────
        function disqualify(field, reason) {
            lead[field] = reason;
            lead.status = 'Disqualified';
            hideProgress();
            showStep('stepDisq');
        }

        // ─────────────────────────────────────────────────────
        // PROGRESS BAR
        // ─────────────────────────────────────────────────────
        function updateProgress(stepNum) {
            const pct = Math.round((stepNum / TOTAL_STEPS) * 100);
            document.getElementById('fill').style.width = pct + '%';
            document.getElementById('stepLabel').textContent = 'Step ' + stepNum + ' of ' + TOTAL_STEPS;
            document.getElementById('stepPct').textContent = pct + '%';
        }

        function hideProgress() {
            document.getElementById('progressWrap').style.display = 'none';
        }

        // ─────────────────────────────────────────────────────
        // SHOW STEP HELPER
        // ─────────────────────────────────────────────────────
        function showStep(id) {
            document.querySelectorAll('.step').forEach(s => s.classList.remove('active'));
            document.getElementById(id).classList.add('active');
        }

        // ─────────────────────────────────────────────────────
        // PHONE FORMATTER
        // ─────────────────────────────────────────────────────
        document.getElementById('phone').addEventListener('input', function(e) {
            let val = e.target.value.replace(/\D/g, '');
            if (val.length >= 6) {
                val = '(' + val.substring(0,3) + ') ' + val.substring(3,6) + '-' + val.substring(6,10);
            } else if (val.length >= 3) {
                val = '(' + val.substring(0,3) + ') ' + val.substring(3);
            }
            e.target.value = val;
        });

        // ─────────────────────────────────────────────────────
        // VALIDATE FIELDS
        // ─────────────────────────────────────────────────────
        function validate() {
            let valid = true;
            const required = ['firstName', 'lastName', 'businessName', 'phone', 'email'];

            required.forEach(id => {
                const el = document.getElementById(id);
                if (!el.value.trim()) {
                    el.classList.add('error-field');
                    valid = false;
                } else {
                    el.classList.remove('error-field');
                }
            });

            // Email format check
            const emailVal = document.getElementById('email').value;
            const emailOk = /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(emailVal);
            if (!emailOk) {
                document.getElementById('email').classList.add('error-field');
                valid = false;
            }

            // TCPA checkbox
            if (!document.getElementById('tcpaConsent').checked) {
                valid = false;
                showError('You must agree to the terms to submit.');
            }

            if (!valid && document.getElementById('tcpaConsent').checked) {
                showError('Please fill in all required fields.');
            }

            return valid;
        }

        function showError(msg) {
            const el = document.getElementById('errorMsg');
            el.textContent = '⚠️ ' + msg;
            el.style.display = 'block';
        }

        function hideError() {
            document.getElementById('errorMsg').style.display = 'none';
        }

        // ─────────────────────────────────────────────────────
        // SUBMIT LEAD TO GOOGLE SHEETS
        // ─────────────────────────────────────────────────────
        async function submitLead() {
            hideError();
            if (!validate()) return;

            // Collect contact fields
            lead.firstName    = document.getElementById('firstName').value.trim();
            lead.lastName     = document.getElementById('lastName').value.trim();
            lead.businessName = document.getElementById('businessName').value.trim();
            lead.phone        = document.getElementById('phone').value.trim();
            lead.email        = document.getElementById('email').value.trim();
            lead.industry     = document.getElementById('industry').value.trim();
            lead.timestamp    = new Date().toISOString();
            lead.source       = document.referrer || 'Direct';
            lead.status       = 'New Lead';

            // Disable button and show loading
            const btn = document.getElementById('submitBtn');
            btn.disabled = true;
            btn.textContent = '⏳ Submitting...';

            try {
                await fetch(SCRIPT_URL, {
                    method:  'POST',
                    mode:    'no-cors',
                    cache:   'no-cache',
                    headers: { 'Content-Type': 'application/json' },
                    body:    JSON.stringify(lead)
                });
            } catch (err) {
                // Still show success — no-cors mode always throws on response
                console.log('Fetch note:', err);
            }

            // Always show success after submit attempt
            hideProgress();
            showStep('stepSuccess');
        }

        // ─────────────────────────────────────────────────────
        // RESET (start over from disq screen)
        // ─────────────────────────────────────────────────────
